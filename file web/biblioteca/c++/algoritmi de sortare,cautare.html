⚙️ Algoritmi clasici în C++

🔹 Căutare liniară

Parcurge vectorul element cu element.

int cauta(int v[], int n, int x) {
    for (int i = 0; i < n; i++)
        if (v[i] == x)
            return i; // găsit la poziția i
    return -1; // nu există
}

🔹 Căutare binară (vector sortat)

int cautaBinar(int v[], int n, int x) {
    int st = 0, dr = n - 1;
    while (st <= dr) {
        int m = (st + dr) / 2;
        if (v[m] == x) return m;
        else if (x < v[m]) dr = m - 1;
        else st = m + 1;
    }
    return -1;
}

🔹 Sortare prin selecție (Selection Sort)

void selectionSort(int v[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int minIndex = i;
        for (int j = i + 1; j < n; j++)
            if (v[j] < v[minIndex])
                minIndex = j;
        swap(v[i], v[minIndex]);
    }
}

🔹 Sortare prin inserție (Insertion Sort)

void insertionSort(int v[], int n) {
    for (int i = 1; i < n; i++) {
        int x = v[i], j = i - 1;
        while (j >= 0 && v[j] > x) {
            v[j+1] = v[j];
            j--;
        }
        v[j+1] = x;
    }
}

🔹 Sortare rapidă (Quick Sort)

int partitie(int v[], int st, int dr) {
    int pivot = v[dr], i = st - 1;
    for (int j = st; j < dr; j++) {
        if (v[j] <= pivot) {
            i++;
            swap(v[i], v[j]);
        }
    }
    swap(v[i+1], v[dr]);
    return i+1;
}

void quickSort(int v[], int st, int dr) {
    if (st < dr) {
        int p = partitie(v, st, dr);
        quickSort(v, st, p - 1);
        quickSort(v, p + 1, dr);
    }
}
